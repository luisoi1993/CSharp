        [Fact]
        public async Task UpdateSerieAsync_WithValidNameUpdateSameSerie()
        {
            
            var dto = new SerieDTO
            {
                SerieID = 1,
                Name = "SERIE-A",
                ExerciseID = 1,
                SerieCorrection = false
            };

            var existingSeries = new List<SerieDTE>
    {
        new SerieDTE
        {
            SerieID = 1,
            Name = "SERIE-A",
            SerieCorrection = false,
            ExerciseID = 1
        }
    };

            var pagedSeries = new ResultDataDto<SerieDTE>
            {
                data = existingSeries
            };

            var mappedSeries = new List<SerieDTO>
    {
        new SerieDTO
        {
            SerieID = 1,
            Name = "SERIE-A",
            SerieCorrection = false
        }
    };

            var mappedResult = new SerieDTO
            {
                SerieID = 1,
                Name = "SERIE-A",
                ExerciseID = 1,
                SerieCorrection = false
            };

            
            _mockRepo
                .Setup(r => r.GetPagedListAsync<SerieDTE>(
                    It.IsAny<SerieDTF>(),
                    It.IsAny<bool>()))
                .ReturnsAsync(pagedSeries);

            _mockRepo
                .Setup(r => r.HasInvoicesBySerieIdAsync(It.IsAny<long>()))
                .ReturnsAsync(false);

            _mockRepo
                .Setup(r => r.PersistAsync(It.IsAny<SerieDTE>()))
                .ReturnsAsync(existingSeries.First());

            
            _mockMapper
                .Setup(m => m.Map(
                    It.IsAny<IEnumerable<SerieDTE>>(),  
                    It.IsAny<List<SerieDTO>>()))
                .Returns(mappedSeries);

            _mockMapper
                .Setup(m => m.Map(
                    It.IsAny<SerieDTE>(),
                    It.IsAny<SerieDTO>()))
                .Returns(mappedResult);

            
            var result = await _serieService.UpdateSerieAsync(dto);

            
            Assert.NotNull(result);
            Assert.Equal(dto.SerieID, result.SerieID);
            Assert.Equal(dto.Name, result.Name);

           
            _mockRepo.Verify(r => r.GetPagedListAsync<SerieDTE>(
                It.IsAny<SerieDTF>(),
                It.IsAny<bool>()),
                Times.Once);
            _mockRepo.Verify(r => r.HasInvoicesBySerieIdAsync(1), Times.Once);
            _mockRepo.Verify(r => r.HasRecurringInvoicesBySerieAsync(It.IsAny<string>()), Times.Never);
        }
