                            /*
                     * Test unitario que verifica el comportamiento de
                     * GetCalculatedConceptDTEListAsync cuando una factura
                     * no tiene conceptos asociados.
                     *
                     * ¿Qué comprueba?
                     * - Que el servicio busque correctamente los conceptos
                     *   asociados a una factura mediante su InvoiceIssuedID.
                     *
                     * - Que, si el repositorio devuelve una lista vacía,
                     *   el servicio no intente calcular impuestos,
                     *   importes base ni retenciones.
                     *
                     * - Que el método devuelva una lista vacía correctamente
                     *   inicializada, en lugar de devolver null o lanzar excepción.
                     *
                     * ¿Cómo funciona?
                     *
                     * 1. Se crea una factura de prueba (InvoiceIssuedDTE)
                     *    con un InvoiceIssuedID = 1.
                     *
                     *    Este ID representa la factura sobre la que
                     *    queremos buscar conceptos.
                     *
                     * 2. Se mockea el repositorio (_mockRepo) para simular
                     *    que al buscar BillConcept asociados a esa factura,
                     *    la base de datos no devuelve ningún registro:
                     *
                     *      data = []
                     *
                     *    Es decir:
                     *    la factura existe, pero no tiene líneas/conceptos.
                     *
                     * 3. Se ejecuta:
                     *
                     *      GetCalculatedConceptDTEListAsync(...)
                     *
                     *    pasándole:
                     *
                     *      - una empresa (RepCompanyDTE)
                     *      - un participante (ParticipantDTO)
                     *      - la factura creada
                     *
                     * 4. Internamente el servicio:
                     *
                     *      - consulta el repositorio
                     *      - recibe lista vacía
                     *      - comprueba que no hay elementos
                     *      - NO entra en CalculateBillConcepts()
                     *      - devuelve una lista vacía
                     *
                     * 5. Finalmente se valida:
                     *
                     *      Assert.NotNull(result)
                     *
                     *    para comprobar que el resultado existe
                     *    y no devuelve null.
                     *
                     *    y:
                     *
                     *      Assert.Empty(result)
                     *
                     *    para comprobar que la lista está vacía.
                     *
                     * Resultado esperado:
                     * - No debe lanzar excepción.
                     * - No debe devolver null.
                     * - Debe devolver una lista vacía ([]).
                     *
                     * Esto garantiza que el servicio maneja correctamente
                     * el caso base en el que una factura todavía no tiene
                     * conceptos asociados.
                     */

        [Fact]
        public async Task GetCalculatedConceptDTEListAsync_WithNoConcepts_ShouldReturnEmptyList()
        {
            
            var invoice = new InvoiceIssuedDTE { InvoiceIssuedID = 1 };

            _mockRepo
                .Setup(r => r.GetPagedListAsync<BillConceptDTE>(
                    It.IsAny<BillConceptDTF>(),
                    It.IsAny<bool>()))
                .ReturnsAsync(new ResultDataDto<BillConceptDTE>
                {
                    data = new List<BillConceptDTE>()
                });

            
            var result = await _service.GetCalculatedConceptDTEListAsync(
                new RepCompanyDTE(),
                new ParticipantDTO(),
                invoice
            );

            
            Assert.NotNull(result);
            Assert.Empty(result);
        }

        /*
         * Verifica que cuando existen conceptos,
         * el servicio los devuelve y ejecuta el cálculo.
         */
        [Fact]
        public async Task GetCalculatedConceptDTEListAsync_WithConcepts_ShouldCalculateAndReturnList()
        {
            
            var invoice = new InvoiceIssuedDTE { InvoiceIssuedID = 1 };

            var concepts = new List<BillConceptDTE>
            {
                new BillConceptDTE
                {
                    Units = 2,
                    Price = 100,
                    PercentDiscount = 0,
                    VATID = 1,
                    Type = 0
                }
            };

            _mockRepo
                .Setup(r => r.GetPagedListAsync<BillConceptDTE>(
                    It.IsAny<BillConceptDTF>(),
                    It.IsAny<bool>()))
                .ReturnsAsync(new ResultDataDto<BillConceptDTE>
                {
                    data = concepts
                });

            _mockTaxService
                .Setup(t => t.GetByIdAsync<TaxDTE>(It.IsAny<long?>()))
                .ReturnsAsync(new TaxDTE { PercentTax = 21 });

            _mockTaxService
                .Setup(t => t.GetByUidAsync<TaxDTE>(It.IsAny<string>()))
                .ReturnsAsync((TaxDTE)null);

            _mockConceptService
                .Setup(c => c.GetByIdAsync<ConceptDTE>(It.IsAny<long?>()))
                .ReturnsAsync(new ConceptDTE());

            
            var result = await _service.GetCalculatedConceptDTEListAsync(
                new RepCompanyDTE(),
                new ParticipantDTO(),
                invoice
            );

            
            Assert.NotNull(result);
            Assert.Single(result);
            Assert.NotNull(result.First().Amount);
        }

        /*
* Verifica cálculo de IVA correctamente
*/
        [Fact]
        public async Task GetCalculatedConceptDTEListAsync_WithVAT_ShouldCalculateVATCorrectly()
        {
            
            var invoice = new InvoiceIssuedDTE { InvoiceIssuedID = 1 };

            var concept = new BillConceptDTE
            {
                Units = 2,
                Price = 50,
                PercentDiscount = 0,
                VATID = 1,
                Type = 0
            };

            _mockRepo
                .Setup(r => r.GetPagedListAsync<BillConceptDTE>(
                    It.IsAny<BillConceptDTF>(),
                    It.IsAny<bool>()))
                .ReturnsAsync(new ResultDataDto<BillConceptDTE>
                {
                    data = new List<BillConceptDTE> { concept }
                });

            _mockTaxService
                .Setup(t => t.GetByIdAsync<TaxDTE>(It.IsAny<long?>()))
                .ReturnsAsync(new TaxDTE { PercentTax = 21, Ambit = 1 });

            _mockConceptService
                .Setup(c => c.GetByIdAsync<ConceptDTE>(It.IsAny<long?>()))
                .ReturnsAsync(new ConceptDTE());

            _mockTaxService
                .Setup(t => t.GetByUidAsync<TaxDTE>(It.IsAny<string>()))
                .ReturnsAsync((TaxDTE)null);

            
            var result = await _service.GetCalculatedConceptDTEListAsync(
                new RepCompanyDTE(),
                new ParticipantDTO(),
                invoice
            );

            
            var item = result.First();

            Assert.Equal(100, item.BaseAmount);       // 2*50
            Assert.Equal(21, item.VATPercent);
            Assert.Equal(21, item.VATAmount);         // 100 * 21%
        }
